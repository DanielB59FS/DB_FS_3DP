124 // Define the template class expander GProxy.
125 // This is a core pillar used in the C++11 modernized version of Gateware.
126 // A GProxy is a smart wrapper around a Gateware Interface & Implementation.
127 // The proxy allows safe transfer and allocation/deallocation of interfaces as well as safe access.
128 // Proxy rules: Copying creates safe handles but does not transfer ownership. Only moving r-values can transfer ownership.
129 template<typename _Interface, typename _Implementation, typename... _CreationParams>
130 class GProxy
131 {
132 protected:
133 typedef _Interface forceRaw; // used to support the rare need of construction from raw pointers (prefer to avoid)
134 typedef _Implementation forceStatic; // used to support static calls from proxy ops
135 typedef std::shared_ptr<const _Interface> forceConst; // used to support const correctness in proxy ops
136 typedef GProxy<_Interface, _Implementation, _CreationParams...> thisClass; // short hand for this type of class
137 std::weak_ptr<_Interface> access; // most common way of access via proxy, does not require ownership
138 std::shared_ptr<_Interface> storage; // Hard handle to data, not all instances use this, most are weak by default
139
140 private:
141 // All Proxys are friends of one another (required for casting)
142 template<typename A, typename B, typename... C> friend class GProxy;
143 // This function is what actually makes a GProxy "proxy" an interface & associated implementation.
144 template<typename... _anyArgs>
145 GReturn allocate(_anyArgs&&... _argTransfer)
146 {
147 GReturn result = GReturn::FAILURE;
148 access.reset(); // clear any prior access, storage will be overwritten below
149 storage = std::make_shared<_Implementation>(); // so we can write the below more safely...
150 // *** A COMPILER ERROR HERE MEANS YOU ARE NOT PASSING ".Create(" A VALID SET OF ARGUMENTS ***
151 // *** READ OUR INCLUDED DOCUMENTATION TO FIND THE PROPER ARGUMENTS TO ANY GATEWARE FUNCTION ***
152 if (storage && G_PASS(result = dynamic_cast<_Implementation*>
153 (storage.get())->Create(std::forward<_anyArgs>(_argTransfer)...)))
154 access = storage; // used for primary access
155 else // if we failed to create, destruct the interface immediately
156 storage.reset();
157 // return what happened
158 return result;
159 }
160
161 public:
162 // Creates an invalid but safe handle
163 GProxy() = default;
164 // Special constructor, allows creation from existing implementation.
165 // This should only be used in conjunction with std::shared_from_this from inside an implementation.
166 GProxy(const std::shared_ptr<_Interface>& _sharedFromThis)
167 {
168 access = storage = _sharedFromThis;
169 }
170 // copy constructor
171 GProxy(const thisClass& _cpy)
172 {
173 if (this != &_cpy)
174 access = _cpy.access;
175 }
176 // template version, attempts to convert incoming data to this class type using dynamic_pointer_cast if possible
177 // We use SFINAE to ensure this is only called for non-derived types, those should use the standard overloads
178 template<typename T, typename = std::enable_if<!std::is_convertible<T*, thisClass*>::value>>
179 GProxy(const T& _cpy)
180 {
181 access = std::dynamic_pointer_cast<_Interface>(_cpy.access.lock());
182 }
183 // move constructor does a direct transfer since it is not considered a copy
184 GProxy(thisClass&& _mov) noexcept //: storage(_mov.storage), access(_mov.access)
185 {
186 storage = std::move(_mov.storage);
187 access = std::move(_mov.access);
188 }
189 // template version, attempts to convert incoming data to this class type using dynamic_pointer_cast if possible
190 // We use SFINAE to ensure this is only called for non-derived types, those should use the standard overloads
191 template<typename T, typename = std::enable_if<!std::is_convertible<T*, thisClass*>::value>>
192 GProxy(T&& _mov) noexcept //: storage(std::dynamic_pointer_cast<_Interface>(std::move(_mov.storage))), access(storage)
193 {
194 storage = std::dynamic_pointer_cast<_Interface>(std::move(_mov.storage));
195 access = (storage) ? storage : std::dynamic_pointer_cast<_Interface>(_mov.access.lock());
196 }
197 // Assignment operator
198 virtual thisClass& operator=(const thisClass& _cpy) final
199 {
200 if (this != &_cpy)
201 {
202 access = _cpy.access;
203 storage.reset(); // copy assignment may not transfer or retain any prior ownership
204 }
205 return *this;
206 }
207 // template version, attempts to convert incoming data to this class type using dynamic_pointer_cast if possible
208 // We use SFINAE to ensure this is only called for non-derived types, those should use the standard overloads
209 template<typename T, typename = std::enable_if<!std::is_convertible<T*, thisClass*>::value>>
210 thisClass& operator=(const T& _cpy)
211 {
212 access = std::dynamic_pointer_cast<_Interface>(_cpy.access.lock());
213 storage.reset(); // copy assignment may not transfer or retain any prior ownership
214 return *this;
215 }
216 // move assignment
217 virtual thisClass& operator=(thisClass&& _mov) noexcept final
218 {
219 storage = std::move(_mov.storage);
220 access = std::move(_mov.access);
221 return *this;
222 }
223 // template version, attempts to convert incoming data to this class type using dynamic_pointer_cast if possible
224 // We use SFINAE to ensure this is only called for non-derived types, those should use the standard overloads
225 template<typename T, typename = std::enable_if<!std::is_convertible<T*, thisClass*>::value>>
226 thisClass& operator=(T&& _mov) noexcept
227 {
228 storage = std::dynamic_pointer_cast<_Interface>(std::move(_mov.storage));
229 access = (storage) ? storage : std::dynamic_pointer_cast<_Interface>(_mov.access.lock());
230 return *this;
231 }
232 // basic destructor
233 ~GProxy() = default;
234 // Allows testing the class for validity
235 virtual operator bool() const final { return !access.expired(); };
236 // nullptr assignment is how to clear a proxy before waiting for it to fall out of scope
237 virtual thisClass& operator =(std::nullptr_t) final { access.reset(); storage.reset(); return *this; };
238 // The GProxy varadic template should match the main internal implementation "Create" function.
239 // Though the templated variadic version of this function can support any set of arguments,
240 // This version requires derived classes to define & support at least one specific Create variant.
241 // This is required for API consistency. Alternate Create routines are supported by the next function.
242 virtual GReturn Create(_CreationParams... _parameter) final
243 {
244 return allocate<_CreationParams...>(std::forward<_CreationParams>(_parameter)...);
245 }
246 // Provided for alternate Create function potential.(direct use by derived classes is optional)
247 // This version is much more functional but cannot be virtual which is why the prior version exists.
248 template< typename... _AltArguments,
249 typename = std::enable_if<!std::is_same< // SFINAE used to force virtual override in impl.
250 std::tuple<_AltArguments...>, std::tuple<_CreationParams...> >::value>>
251 GReturn Create(_AltArguments... _parameter)
252 {
253 return allocate<_AltArguments...>(std::forward<_AltArguments>(_parameter)...);
254 }
255 // Return a copy of the Proxy that allows for non-weak transfer of ownership
256 // If direct ownership cannot be shared it will return an invalid handle
257 virtual thisClass Share() const final
258 {
259 thisClass canShare;
260 if (storage)
261 {
262 canShare.storage = storage;
263 canShare.access = access;
264 }
265 return canShare; // r-value will invoke move to allow full transfer
266 }
267 // Surrenders strong ownership of your internal handle to another Proxy, weak ownership is maintained
268 // If direct ownership cannot be surrendered it will return an invalid handle
269 virtual thisClass Relinquish() final
270 {
271 thisClass canYield;
272 if (storage)
273 {
274 canYield.storage = storage;
275 canYield.access = access;
276 storage.reset();
277 }
278 return canYield;
279 }
280 // Returns a direct copy of the Proxy with the current ownership status maintined
281 // If direct ownership cannot be shared it will provide it's weak ownership
282 virtual thisClass Mirror() const final
283 {
284 thisClass canReflect;
285 canReflect.storage = storage;
286 canReflect.access = access;
287 return canReflect;
288 }

405 // Defines the simple type-erasure struct GEvent
406 // This is a another core pillar used in the C++11 modernized version of Gateware.
407 // A GEvent is a generic but type-safe way to communicate small-footprint event messages to listeners.
408 // All sent messages must fit within G_MESSAGE_BLOCK_SIZE total including type_info hashes.
409 // A GEvent only supports safe transfer of raw data blocks(C structs, POD) & enum class types.
410 // (Copy)Constructors/Destructors/Assignment Operators etc... of stored types are NOT respected in any way!
411 // Future versions (c++17+) of Gateware could replace some of this with the FAR more robust std::any if needed.
412
413 //! Typesafe and Typeless communication of simplistic event data. Used by GEventGenerator & GEventReceiver. (Observer pattern)
414 /*!
415 * This struct is how GEventGenerators communicate custom Events to any GEventReceiver who registers themselves to the object.
416 * This struct alongside std::function/std::bind is used to circumvent the traditional derivation based observer model.
417 * You can think of GEvent as a less-robust and more specialized version of std::any. (we currently only target c++11)
418 */
419 struct GEvent
420 {
421 private:
422 std::size_t enumType = 0; // identifies stored enumerator representing the event ID
423 std::size_t dataType = 0; // identifies stored structure representing the associated data bundled with an event
424 // raw data containing specific enum value & data packet.
425 // The leading byte of each type-erased data block records the size of the embedded data.
426 // This information is used for additional data integrity verification during reads and offset jumps when required.
427 unsigned char rawData[G_RAW_DATA_PACKET_SIZE] = {0,};
428 // due to odd include order issues across platforms regarding std::memcpy & std::copy
429 // we provide a substitution here that copies a fixed amount of bytes
430 static inline void memory_copy(void *_dst, const void* _src, unsigned int _byteCount)
431 {
432 std::size_t* write = static_cast<std::size_t*>(_dst);
433 std::size_t const *read = static_cast<std::size_t const*>(_src);
434 for (; _byteCount >= sizeof(std::size_t); _byteCount -= sizeof(std::size_t))
435 *write++ = *read++;
436 unsigned char* w = reinterpret_cast<unsigned char*>(write);
437 unsigned char const* r = reinterpret_cast<unsigned char const*>(read);
438 for (; _byteCount; --_byteCount)
439 *w++ = *r++;
440 }
441
442 public:
443 //! Allows writing of any enum class & raw data to a GEvent while retaining internal type hashes.(typeerasure)
444 /*!
445 * DEVNOTE: Written event data must be <= to (G_MESSAGE_BLOCK_SIZE - 16) - sizeof(chosen enum class type).
446 * Only base type enums & raw data streams (C style structs) may be written as data (POD Plain Old Data)
447 * Each stored type may not exceed 255 bytes for size verification/offset reasons.
448 *
449 * \param _eventEnumValue Enum used to represent which event has occured.(any simple POD enum valid)
450 * \param _eventData Extra data that acompanies this event.(any small POD works here)
451 *
452 * \retval SUCCESS Valid types were provided and the event data was successfully embedded.
453 * \retval UNEXPECTED_RESULT Encoded type matched the provided type hash but does not match its size in bytes.
454 */
455 template<class enumT, typename dataT>
456 GReturn Write(const enumT& _eventEnumValue, const dataT& _eventData)
457 {
458 // ensure developers stay within data size limits of storable data types
459 static_assert( sizeof(enumT) < 256, "Enum types stored by GEvent must be < 256 bytes!");
460 static_assert( sizeof(dataT) < 256, "Data types stored by GEvent must be < 256 bytes!");
461 // ensure developers do not exceed internal storage limits
462 static_assert( (sizeof(enumT) + sizeof(dataT)) <= G_RAW_DATA_PACKET_SIZE,
463 "Enum + Data combined size attempted to be stored by GEvent is too large!");
464 // ensure developers do not send types other than an enum to identify their events
465 static_assert( std::is_enum<enumT>::value,
466 "Gateware Events{} are required to use an \"enum class\" list to identify themselves!");
467 // ensure developers use only C Style stuctures & basic enumerators for information transfer
468 static_assert( std::is_trivially_copyable<dataT>::value,
469 "Enum & Data Types must be fully tranferable with std:memcpy only! Plain Old Data Only!");
470 // Once we know the storable types are safe we record their type and transfer the data
471 enumType = typeid(enumT).hash_code();
472 dataType = typeid(dataT).hash_code();
473 // Unlikely, but the standard doesn't say it can't happen, would interfere with error checking
474 if (enumType == 0 || dataType == 0)
475 return GReturn::UNEXPECTED_RESULT;
476 // Store bit data for typeless transfer
477 rawData[0] = sizeof(enumT); // encode size of enum
478 memory_copy(rawData + 1, &_eventEnumValue, sizeof(enumT)); // store enum
479 rawData[sizeof(enumT) + 1] = sizeof(dataT); // encode size of data
480 memory_copy(rawData + sizeof(enumT) + 2, &_eventData, sizeof(dataT)); // store data
481 // transfer to internal storage should now be successful
482 return GReturn::SUCCESS;
483 };
484 //! Safely extract G::Events and G::EVENT_DATA from a GEventGenerator if you know what type of information you need.
485 /*!
486 * Once your GEventReceiver has responded to an event and is receiving a GEvent, you can use this operation
487 * to safely extract information about a specific event. Not only can you establish which event was generated,
488 * you can also receive detailed information about the event if that event contains additional data.
489 *
490 * \param [out] _outEventID Variable used to capture which event has occured.(See interface docs)
491 * \param [out] _outEventData Variable used to capture incoming data from an event.(See interface docs)
492 *
493 * \retval SUCCESS The proper types were provided and the event data was successfully extracted.
494 * \retval FAILURE This GEvent was never intialized with any event data. (Dataless Event)
495 * \retval INVALID_ARGUMENT Provided template data types do NOT match internally stored hashes for this event.
496 * \retval MEMORY_CORRUPTION Encoded type matched the provided type hash but does not match its size in bytes.
497 */
498 template<class enumT, typename dataT>
499 GReturn Read(enumT& _outEventID, dataT& _outEventData) const
500 {
501 // If you get this complier error you are providing Read() with an incompatible type.
502 // Always use Gxxx::Events and Gxxx::EVENT_DATA to examine an event.
503 static_assert((sizeof(enumT) + sizeof(dataT)) <= G_RAW_DATA_PACKET_SIZE &&
504 std::is_enum<enumT>::value && std::is_trivially_copyable<enumT>::value &&
505 std::is_trivially_copyable<dataT>::value,
506 "Types provided to GEvent::Read() operation are not a compatible types!"
507 "Use appropriate GEventGenerator derived G::Events enum and G::EVENT_DATA structure.");
508 // Type hashes will ensure safe access of underlying data in 99.9% of situations
509 // Not 100% perfect but much safer than a void* like it used to be.
510 if (enumType == 0 || dataType == 0) return GReturn::FAILURE; // No event data present
511 // Requested type does not match internal enum format (use proper G::Events)
512 if (enumType != typeid(enumT).hash_code()) return GReturn::INVALID_ARGUMENT;
513 // Requested type does not match internal data structure format (use proper G::EVENT_DATA)
514 if (dataType != typeid(dataT).hash_code()) return GReturn::INVALID_ARGUMENT;
515 // Once types have been safely checked we verify the byte sizes of types and copy data
516 if (sizeof(enumT) == rawData[0])
517 memory_copy(&_outEventID, rawData + 1, sizeof(enumT));
518 else
519 return GReturn::MEMORY_CORRUPTION; // data size verification failed
520 // check that data type is also size verified before transfer
521 if (sizeof(dataT) == rawData[sizeof(enumT) + 1])
522 memory_copy(&_outEventData, rawData + sizeof(enumT) + 2, sizeof(dataT));
523 else
524 return GReturn::MEMORY_CORRUPTION; // data size verification failed
525 // if we made it this far everything checked out and transfered successfully.
526 return GReturn::SUCCESS;
527 }
528 //! Safely extract G::Events and G::EVENT_DATA from a GEventGenerator if you know what type of information you need.
529 /*!
530 * Once your GEventReceiver has responded to an event and is receiving a GEvent, you can use this operation
531 * to safely extract information about a specific event. Not only can you establish which event was generated,
532 * you can also receive detailed information about the event if that event contains additional data.
533 *
534 * \param [out] _outEventOrData Variable used to capture incoming data or event ID from an event.(See interface docs)
535 *
536 * \retval SUCCESS A proper type was provided and the information was successfully extracted.
537 * \retval FAILURE This GEvent was never intialized with any information. (Uninitialized Event)
538 * \retval INVALID_ARGUMENT Provided template data type does NOT match any internal stored hash for this event.
539 * \retval MEMORY_CORRUPTION Encoded type matched the provided type hash but does not match its size in bytes.
540 */
541 template<class enumOrDataT>
542 GReturn Read(enumOrDataT& _outEventOrData) const
543 {
544 // If you get this complier error you are providing Read() with an incompatible type.
545 // Always use Gxxx::Events and Gxxx::EVENT_DATA to examine an event.
546 static_assert(sizeof(enumOrDataT) <= G_RAW_DATA_PACKET_SIZE &&
547 std::is_trivially_copyable<enumOrDataT>::value,
548 "Type provided to GEvent::Read() operation is not a compatible type!"
549 "Use appropriate GEventGenerator derived G::Events enum or G::EVENT_DATA structure.");
550 // Type hashes will ensure safe access of underlying data in 99.9% of situations
551 // Not 100% perfect but much safer than a void* like it used to be.
552 if (enumType == 0 && dataType == 0)
553 return GReturn::FAILURE; // No event data present
554 // enum type identified
555 else if (enumType == typeid(enumOrDataT).hash_code())
556 if (sizeof(enumOrDataT) == rawData[0])
557 memory_copy(&_outEventOrData, rawData + 1, sizeof(enumOrDataT)); // transfer enum bits
558 else // data size verification failed
559 return GReturn::MEMORY_CORRUPTION;
560 // event data identified
561 else if (dataType == typeid(enumOrDataT).hash_code())
562 if (rawData[0] < (G_EVENT_BLOCK_SIZE - (sizeof(std::size_t) << 1)) && // enoded enum type integrity check
563 sizeof(enumOrDataT) == rawData[rawData[0] + 1]) // jump to type size offset
564 memory_copy(&_outEventOrData, rawData + rawData[0] + 2, sizeof(enumOrDataT)); // transfer enum bits
565 else // data size verification failed
566 return GReturn::MEMORY_CORRUPTION;
567 else // Requested type does not match internal enum/data structure format (use proper G::Events/G::EVENT_DATA)
568 return GReturn::INVALID_ARGUMENT;
569 // with transfer complete notify success
570 return GReturn::SUCCESS;
571 }
572 };

661// GATEWARE_TEMPLATE_FUNCTION(GATEWARE_DEPRECATED(Foo, FoosReplacement) Foo)
